pipeline {
    agent {
        docker { image 'node:16-alpine' }
    }
    environment {
        GIT_REPO = 'https://github.com/shyam9307/Angular-live-project.git'
        APP_DIR = 'angular-app'
    }
    stages {
        stage('Clone Repository') {
            steps {
                sh 'git clone $GIT_REPO $APP_DIR'
            }
        }
        stage('Install Dependencies') {
            steps {
                sh 'cd $APP_DIR && npm install'
            }
        }
        stage('Start Application') {
            steps {
                sh 'cd $APP_DIR && npm start'
            }
        }
    }
}
